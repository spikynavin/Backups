#include <Keypad.h>
#include <EEPROM.h>
#include <DS3231.h>
#include <Wire.h>
#include <LiquidCrystal_I2C.h>

/* include the above libraries */

DS3231 rtc(SDA, SCL); //RTC Modules pin
Time t;
LiquidCrystal_I2C lcd(0x27,20,4); //LCD I2C ID
int Hor, Min, Sec, tim, dat, h, m, s;

void setup(){
    Serial.begin(115200);
    Wire.begin();
    rtc.begin();
    lcd.init();
    lcd.backlight();
    timedate();

    //uncomment these lines to set the date and time
    //rtc.setDOW(MONDAY); // Set Day-of-Week to SUNDAY
    //rtc.setTime(10, 06, 0); // Set the time to 12:00:00 (24hr format)
    //rtc.setDate(27, 1, 2020); // Day, Month, Year
}

void loop(){
    t = rtc.getTime();
    Hor = t.hour;
    Min = t.min;
    Sec = t.sec;
    tim = rtc.getTimeStr();
    dat = rtc.getDateStr();
    lcd.setCursor(0,0);
    lcd.print("Timer:");
    lcd.print(rtc.getTimeStr());
}

void timedate(){
lcd.clear();
lcd.setCursor(0,0);
lcd.print("Time: ");
lcd.print(rtc.getTimeStr());
lcd.setCursor(0,1);
lcd.print("Date: ");
lcd.print(rtc.getDateStr());
lcd.setCursor(0,20);
lcd.display();
}
